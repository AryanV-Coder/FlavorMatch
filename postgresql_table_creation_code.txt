## FAMILY TABLE
create table public.family (
  family_id bigint generated by default as identity not null,
  family_username character varying not null,
  constraint family_pkey primary key (family_id),
  constraint family_family_username_key unique (family_username)
) TABLESPACE pg_default;

## MEMBER TABLE
create table public.member (
  member_id bigint generated by default as identity not null,
  family_id bigint not null,
  member_username character varying not null,
  member_password character varying not null,
  constraint member_pkey primary key (member_id),
  constraint member_family_username_unique unique (family_id, member_username),
  constraint member_family_id_fkey foreign KEY (family_id) references family(family_id) on update CASCADE on delete CASCADE
) TABLESPACE pg_default;

## FOOD TABLE
create table public.food (
  food_id bigint generated by default as identity not null,
  timestamp timestamp without time zone not null,
  member_id bigint not null,
  food character varying not null,
  is_liked character varying not null,
  is_healthy character varying not null,
  constraint food_pkey primary key (food_id),
  constraint food_member_id_fkey foreign KEY (member_id) references member (member_id) on update CASCADE on delete CASCADE
) TABLESPACE pg_default;
